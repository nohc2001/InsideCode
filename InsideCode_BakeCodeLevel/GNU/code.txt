struct vec3f{
	float x;
	float y;
	float z;
};

void print_sentence(char* str, int len){
	int i=0;while(i<len){
		dbg(str[i]);
		i += 1;
	}
}

void print_vec3f(vec3f obj){
	print_sentence("vector = (");
	dbg(obj.x);
	print_sentence(", ");
	dbg(obj.y);
	print_sentence(", ");
	dbg(obj.z);
	print_sentence(")\n");
}

int main(){
	vec3f A;
	A.x = 10;
	A.y = 20;
	A.z = 30;
	print_vec3f(A);
	return 0;
}
